# 使用例子

# 一. Docker 创建容器

  1. 直接创建
创建一个伪终端时刻保持连接，并使用bash作为交互命令
docker run -itd -p 8088:8080 node bash

# 二. Dockerfile 创建镜像

一般结合第三种一起使用

  1. docker build -t ng:v5 .
FROM nginx
RUN apt-get update
COPY a.txt /www/ 
ADD a.txt /www/  
ENV NODE_VERSION 7.2.0
EXPOSE []
WORKDIR
CMD ["php fpm","<param1>","<param2>",...] 
# 定义匿名数据卷。在启动容器时忘记挂载数据卷，会自动挂载到匿名卷。
VOLUME [""]

# 三. 使用docker-compose 创建容器

version: "3" #版本
services: #服务
  nginx: #nginx服务
    build: . #构建的dockerfile位置
    image: nginx #构建镜像名称 和build不可同时存在
    container_name: "nginx_common" #构建容器名称
    networks: #使用的网络 不设置则默认分配
    ports: #端口映射关系
    depends_on: #依赖关系 无可不设置
    links: #链接容器 无可不设置
    volumes: #共享目录

简单例子

version: "3"
services:
  nginx:
    build: ./nginx
    container_name: "nginx_common"
    networks:
      - "net1"
    ports:
      - "80:80"
    depends_on:
      - "php71"
    links:
      - "php71"
    volumes:
      - ./www:/www
  php71:
    build: ./php71
    container_name: "php7.1"
    networks:
      - "net1"
    ports:
      - "8002:8002"
    volumes:
      - ./www:/www
networks:
  net1:
    driver: bridge
